import numpy as np
from sklearn.linear_model import LogisticRegression

def train(X, y):
    clf = LogisticRegression()
    return clf.fit(X, y)

def predict(clf, X):
    return clf.predict(X)

X_train = np.array([[1], [2], [3]])
y_train = np.array([0, 0, 1])

clf = train(X_train, y_train)
print(predict(clf, np.array([[4]])))
